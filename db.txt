CREATE TABLE patient (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
    ic VARCHAR(20),
    email VARCHAR(50),
    mobile VARCHAR(20),
    gender VARCHAR(10),
    address TEXT,
    firstName VARCHAR(255),
    lastName VARCHAR(255),
    photo VARCHAR(255)
);

CREATE TABLE dentist (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
    ic VARCHAR(20),
    email VARCHAR(50),
    mobile VARCHAR(20),
    gender VARCHAR(10),
    address TEXT,
    password VARCHAR(255),
    firstName VARCHAR(255),
    lastName VARCHAR(255),
    photo VARCHAR(255),
    isAdmin bool
);

CREATE TABLE nurse (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
    ic VARCHAR(20),
    email VARCHAR(50),
    mobile VARCHAR(20),
    gender VARCHAR(10),
    address TEXT,
    password VARCHAR(255),
    firstName VARCHAR(255),
    lastName VARCHAR(255),
    photo VARCHAR(255),
    isAdmin bool
);

CREATE TABLE review (
    review_id INT AUTO_INCREMENT PRIMARY KEY,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    rating INT,
    comments TEXT,
    date DATE,
    patient_id INT, 
    FOREIGN KEY (patient_id) REFERENCES patient(id) 
);

CREATE TABLE user (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
    ic VARCHAR(20),
    email VARCHAR(50),
    mobile VARCHAR(20),
    gender VARCHAR(10),
    address TEXT,
    password VARCHAR(255),
    firstName VARCHAR(255),
    lastName VARCHAR(255),
    photo VARCHAR(255)
);

CREATE TABLE schedule (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
    dentistID INT,
    nurseID INT,
    date DATE,
    time_from TIME,
    time_to TIME,
    room_number VARCHAR(10),
    update_date DATE,
    FOREIGN KEY (dentistID) REFERENCES dentist(id),
    FOREIGN KEY (nurseID) REFERENCES nurse(id)
);
